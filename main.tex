% Define verse counters
\newcounter{versea}

\begin{document}

%Command at the start of chapter
	\NewDocumentCommand{\chapterStart}{o O{Chap}}{%
		\setcounter{versea}{0}%
		\setcounter{verseb}{0}%
		\stepcounter{chapter}%
		\IfNoValueF{#1}{%
			\begin{center}%
			\textbf{#2. \arabic{chapter}} \\
			{#1}\end{center}%
		}%
	}
	
%Command at the start of book
	\newcommand{\bookStart}{%
		\setcounter{chapter}{0} % Set chapter count to zero.
		\chapterStart{}%
	}

% Verse format commands
	\NewDocumentCommand{\bva}{o}{% Begin verse a
		\stepcounter{versea}%
		\noindent% No indentation for the verse
		\IfNoValueF{#1}{%
		\ifnum\value{versea}>1 % If it is the first verse, we do not show it.
			\textsuperscript{\arabic{versea}}\fi % Styling. The verse is just the number in superscript. 
		}{%
			\textsuperscript{\arabic{versea}}%
		}%
	}
	\NewDocumentCommand{\eva}{o}{% End verse a
		\\%
	}
	
	\NewDocumentCommand{\bvb}{o}{% Begin verse b
		\stepcounter{verseb}%
		\IfNoValueF{#1}{%
		\ifnum\value{verseb}>1 % If it is the first verse, we do not show it.
			\textsuperscript{\arabic{verseb}}\fi % Styling. The verse is just the number in superscript. 
		}{%
			\textsuperscript{\arabic{verseb}}%
		}%
	}
	\NewDocumentCommand{\evb}{o}{% End verse b
		\\%
	}
	
% Poem formatting
	% No left-indentation.
	\setlength{\vleftmargin}{0em}

	% Mark cæsura.
	\newcommand{\hld}{\hspace{5mm} }%\leavevmode\unskip\quad\ignorespaces}

	% Indent lines (in Ljóðaháttr or Galdralag).

		\hspace{1.5em}%
	}

	% Mark alliteration. This might not be present in the final version.
	\NewDocumentCommand{\alst}{m}{%
		\underline{#1}%
	}
